Integrasi Google Apps Script (GAS)

Langkah menghubungkan form ke Google Sheet Anda

Jika Anda tidak menanamkan URL langsung pada script utama HTML, Anda dapat mengaturnya sementara di kolom input di atas.

Langkah Setup Backend & Template Google Sheet Baru

  1. Buat sebuah Spreadsheet Baru di Google Drive Anda.
  2. Beri nama tab pertama: PesananSeragam.
  3. Tuliskan kolom-kolom berikut sebagai header di baris ke-1 (Mulai kolom A sampai N):
    [A] Timestamp [B] No Pendaftaran [C] Nama [D] Asal Sekolah [E] Jenis Kelamin [F] Kelas [G] Baju OSIS [H] Rok/Celana OSIS [I] Batik Kotak [J] Batik Logo [K] Rok/Celana Identitas [L] Baju Pramuka [M] Rok/Celana Pramuka [N] Total Item
  4. Di Spreadsheet Anda, buka menu Ekstensi > Apps Script.
  5. Hapus kode bawaan di dalam editor script, lalu paste kode di bawah ini ke editor Apps Script Anda.
  6. Klik tombol Terapkan / Deploy di sudut kanan atas > pilih Penerapan baru.
  7. Pilih jenis penerapan: Aplikasi Web. Atur Aksesibilitas ke: Siapa saja (Anyone).
  8. Salin URL aplikasi web yang dihasilkan dan tempelkan pada kolom input di atas!

Kode Code.gs

/*
 * SCRIPT GOOGLE APPS SCRIPT (Code.gs)
 * Untuk menyimpan pesanan dengan kolom terpisah menurut masing-masing item seragam
 */

function doPost(e) {
  var sheetName = "PesananSeragam";
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(sheetName);
  
  // Jika sheet belum ada, buat baru otomatis dengan header kolom terpisah
  if (!sheet) {
    sheet = ss.insertSheet(sheetName);
    sheet.appendRow([
      "Timestamp", 
      "No Pendaftaran", 
      "Nama", 
      "Asal Sekolah", 
      "Jenis Kelamin", 
      "Kelas", 
      "Baju OSIS", 
      "Rok/Celana OSIS", 
      "Batik Kotak", 
      "Batik Logo", 
      "Rok/Celana Identitas", 
      "Baju Pramuka", 
      "Rok/Celana Pramuka", 
      "Total Item"
    ]);
    sheet.getRange(1, 1, 1, 14).setFontWeight("bold").setBackground("#dcfce7");
  }
  
  try {
    var data = JSON.parse(e.postData.contents);
    var p = data.pesanan || {};
    
    // Ekstrak ukuran masing-masing item secara terpisah (jika tidak ada/tidak dipesan diisi "-")
    var osisBaju = p.osis_baju ? p.osis_baju.ukuran : "-";
    var osisBawahan = p.osis_bawahan ? p.osis_bawahan.ukuran : "-";
    
    var identitasBatikKotak = p.identitas_batik_kotak ? p.identitas_batik_kotak.ukuran : "-";
    var identitasBatikLogo = p.identitas_batik_logo ? p.identitas_batik_logo.ukuran : "-";
    var identitasBawahan = p.identitas_bawahan ? p.identitas_bawahan.ukuran : "-";
    
    var pramukaBaju = p.pramuka_baju ? p.pramuka_baju.ukuran : "-";
    var pramukaBawahan = p.pramuka_bawahan ? p.pramuka_bawahan.ukuran : "-";
    
    // Hitung total item fisik yang dipesan (nilai bukan "-")
    var totalQty = 0;
    var listUkuran = [osisBaju, osisBawahan, identitasBatikKotak, identitasBatikLogo, identitasBawahan, pramukaBaju, pramukaBawahan];
    listUkuran.forEach(function(ukuran) {
      if (ukuran !== "-") {
        totalQty++;
      }
    });

    // Tulis data ke baris baru spreadsheet (14 Kolom)
    sheet.appendRow([
      new Date(),
      data.no_pendaftaran,
      data.nama,
      data.asal_sekolah,
      data.jenis_kelamin,
      data.kelas,
      osisBaju,
      osisBawahan,
      identitasBatikKotak,
      identitasBatikLogo,
      identitasBawahan,
      pramukaBaju,
      pramukaBawahan,
      totalQty + " Pcs"
    ]);
    
    return ContentService.createTextOutput(JSON.stringify({ 
      "status": "success", 
      "message": "Pesanan Anda berhasil disimpan!" 
    })).setMimeType(ContentService.MimeType.JSON)
       .setHeader('Access-Control-Allow-Origin', '*');
       
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ 
      "status": "error", 
      "message": err.toString() 
    })).setMimeType(ContentService.MimeType.JSON)
       .setHeader('Access-Control-Allow-Origin', '*');
  }
}

// Menangani Preflight Request untuk CORS jika browser menggunakannya
function doOptions(e) {
  return ContentService.createTextOutput("")
    .setMimeType(ContentService.MimeType.TEXT)
    .setHeader('Access-Control-Allow-Origin', '*')
    .setHeader('Access-Control-Allow-Methods', 'POST, GET, OPTIONS')
    .setHeader('Access-Control-Allow-Headers', 'Content-Type');
}
